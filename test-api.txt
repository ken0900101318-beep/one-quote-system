/**
 * 執行所有測試
 */
function runAllTests() {
  Logger.log('===== ONE桌遊報價系統 API 測試開始 =====');
  Logger.log('');
  
  const tests = [
    { name: 'Test 1: 基本連線', fn: test1_basicConnection },
    { name: 'Test 2: 取得價目表', fn: test2_getPriceTable },
    { name: 'Test 3: 取得所有專案', fn: test3_getProjects },
    { name: 'Test 4: 用內部 ID 查詢專案', fn: test4_getProjectById },
    { name: 'Test 5: 用專案編號查詢專案', fn: test5_getProjectByNumber },
    { name: 'Test 6: 新增專案', fn: test6_addProject },
    { name: 'Test 7: 更新專案', fn: test7_updateProject },
    { name: 'Test 8: 刪除專案', fn: test8_deleteProject },
    { name: 'Test 9: 取得分類', fn: test9_getCategories }
  ];
  
  let passed = 0;
  let failed = 0;
  
  tests.forEach(test => {
    Logger.log(`\n執行 ${test.name}...`);
    try {
      const result = test.fn();
      if (result) {
        Logger.log(`✅ ${test.name} 通過`);
        passed++;
      } else {
        Logger.log(`❌ ${test.name} 失敗`);
        failed++;
      }
    } catch (error) {
      Logger.log(`❌ ${test.name} 錯誤: ${error.toString()}`);
      failed++;
    }
    Logger.log('');
  });
  
  Logger.log('===== 測試結果 =====');
  Logger.log(`✅ 通過: ${passed}`);
  Logger.log(`❌ 失敗: ${failed}`);
  Logger.log(`總計: ${tests.length}`);
}
